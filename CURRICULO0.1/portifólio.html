Objetivo cobrar um valor diferente com base na idade do cliente


codigo python


    
    # Sistema desenvolvido por João Vitor Abreu
    
    # Input do valor base do plano e da idade do cliente
    valorBase = float(input("Informe o valor base do plano: "))
    idade = int(input("Informe a idade do cliente: "))
    
    # Variável para armazenar o valor percentual
    percentual = 0
    
    # Estruturas condicionais para determinar o percentual do valor base
    if 0 <= idade < 19:
        percentual = 1.00  # 100% do valor base
    elif 19 <= idade < 29:
        percentual = 1.50  # 150% do valor base
    elif 29 <= idade < 39:
        percentual = 2.25  # 225% do valor base
    elif 39 <= idade < 49:
        percentual = 2.40  # 240% do valor base
    elif 49 <= idade < 59:
        percentual = 3.50  # 350% do valor base
    elif idade >= 59:
        percentual = 6.00  # 600% do valor base
    else:
        print("Idade inválida.")
        exit()  # Encerra o programa se a idade for inválida
    
    # Cálculo do valor mensal
    valorMensal = valorBase * percentual
    
    # Exibição dos resultados
    print("Sistema desenvolvido por João Vitor Abreu")  # Nome do desenvolvedor
    print(f"O valor mensal do plano para a idade {idade} anos é: R${valorMensal:.2f}")  # Valor mensal
